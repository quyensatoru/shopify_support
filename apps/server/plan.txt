async function checkShop(shop) {
    const { domain, access_token, charge_id, subscription_info } = shop;

    if (!access_token || !charge_id || !subscription_info?.title) {
        return null;
    }

    try {
        const charge = await getActualPlanFromShopify(domain, access_token, charge_id);
        const dbSubPlan = subscription_info.title.trim().toLowerCase();
        const realSubPlan = charge.name.trim().toLowerCase();

        const isSamePlan = dbSubPlan.includes(realSubPlan) || realSubPlan.includes(dbSubPlan);
        const isCancelled = charge.status === 'cancelled';

        if ((isCancelled && isSamePlan) || !isSamePlan) {
            return {
                domain,
                charge_id,
                chare_status: charge.status,
                db_plan: subscription_info.title,
                real_plan: charge.name,
                cancelled_on: charge.cancelled_on,
            };
        }
        return null;
    } catch (err) {
        console.log(`Error: checking ${domain}: ${err.message}`);
        return null;
    }
}
